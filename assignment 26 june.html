/* Task 1: All css properties with explanation */

/* 1. display → defines how element is shown: block, inline, flex, none */ display: block; /* 2. position → sets element placement: static, relative, absolute, fixed */ position: absolute; /* 3. top → vertical position (used with position: absolute) */ top: 100%; /* 4. left → horizontal position */ left: 0; /* 5. background-color → sets background color */ background-color: #33688a; /* 6. color → sets text color */ color: white; /* 7. padding → space inside the element */ padding: 10px 15px; /* 8. margin → space outside the element */ margin: 0; /* 9. list-style → removes bullets from list */ list-style: none; /* 10. text-decoration → removes underline or styling from text/links */ text-decoration: none; /* 11. font-family → sets the font type */ font-family: Arial, sans-serif; /* 12. z-index → decides which element stays on top (higher = top) */ z-index: 1; /* 13. :hover → triggers style when user hovers over element */ li:hover { background-color: #1c3b52; }
